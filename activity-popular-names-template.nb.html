GlyphReadyForm <- BabyNames %>%
  group_by(year, name) %>%
  filter(
    name == "Sarah" |
    name == "Rachael" |
    name == "Adam" |
    name == "Daisy" |
    name == "Ana"
    )

  xlab("Year") +
